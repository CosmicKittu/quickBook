ArrayList (Equivalent to Vector)

import java.util.*;

public class Main {
    public static void main(String[] args) {
        // ========== Working Principle ========== //
        // Dynamic array — stores elements in contiguous memory.
        // Allows O(1) random access via index.
        // Insertion/deletion in middle/front is O(n) (shifting needed).
        // Resizes by allocating new memory when capacity is exceeded.
        // ======================================== //

        // ========== Declaration ========== //
        // Creates an empty ArrayList of integers.
        ArrayList<Integer> v = new ArrayList<>();
        // Creates an empty ArrayList of strings.
        ArrayList<String> vs = new ArrayList<>();
        
        // Initializes with given values.
        ArrayList<Integer> v1 = new ArrayList<>(Arrays.asList(1, 2, 3, 4, 5));
        // Copies all elements from v1 to v2.
        ArrayList<Integer> v2 = new ArrayList<>(v1);

        // 2D ArrayList for matrices
        ArrayList<ArrayList<Integer>> mat = new ArrayList<>();

        int m = 3, n = 3;
        // m x n matrix initialized with 0
        for(int i = 0; i < m; i++) {
            mat.add(new ArrayList<>(Collections.nCopies(n, 0)));
        }

        // ========== Adding Elements ========== //
        // Adds 13 to the end of the ArrayList.
        v.add(13);

        // Add 10 to the position (start) of the ArrayList
        v.add(0, 10);

        // Adds all three values at once to the ArrayList
        v.addAll(Arrays.asList(14, 20, 25));

        // ========== Deleting Elements ========== //
        // Delete last element.
        v.remove(v.size() - 1);
        // Clear the whole ArrayList.
        v.clear();
        // Removes element at index 1.
        v.remove(1);
        // Removes specific object (if it exists).
        v.remove(Integer.valueOf(14));

        // ========== Getting Elements ========== //
        // Using get()
        v.get(0);
        // Returns first element value
        v.get(0);
        // Returns last element value
        v.get(v.size() - 1);

        // ========== Other Important Methods ========== //
        // Returns the number of elements in the ArrayList.
        v.size();
        // Checks if the ArrayList is empty
        v.isEmpty(); // return true or false

        // Sorting [Time complexity:O(n log n)]
        // Ascending order (default)
        Collections.sort(v1);
        // Descending order
        Collections.sort(v1, Collections.reverseOrder());
        
        // Reverse ArrayList
        Collections.reverse(v1);

        // get maximum value
        int maxVal = Collections.max(v1);
        // get minimum value
        int minVal = Collections.min(v1);

        // count how many times 5 appears
        int countVal = Collections.frequency(v1, 5);

        // Binary search (needs sorted list) returns index, < 0 if not found
        int index = Collections.binarySearch(v1, 10);
        boolean exists = index >= 0;

        // Iterating ArrayList
        for (int num : v1) {
            System.out.println(num);
        }

        // Swapping two elements
        Collections.swap(v1, 0, 1);
    }
}

Stack

import java.util.*;

public class Main {
    public static void main(String[] args) {
        // ========== Declaration ========== //
        // Creates an empty stack of integers.
        Stack<Integer> st = new Stack<>();
        // Note: In modern Java, ArrayDeque is preferred over Stack
        // Deque<Integer> st = new ArrayDeque<>();

        // ========== Adding Elements ========== //
        // Adds 23 to the top of the stack.
        st.push(23);
        st.push(34);

        // ========== Deleting Elements ========== //
        // Removes the top element from the stack and returns it.
        st.pop();

        // ========== Accessing Elements ========== //
        // Returns the top element of the stack without removing it.
        int top_element = st.peek();

        // ========== Other Important Methods ========== //
        // Checks if the stack is empty.
        st.isEmpty(); // Returns true or false.
        // Returns the number of elements in the stack.
        st.size();

        // Iterating Through Stack (Not generally recommended for stacks, but possible)
        while (!st.isEmpty()) {
            System.out.println(st.peek());
            st.pop();
        }
    }
}
03 — JAVA

Queue

import java.util.*;

public class Main {
    public static void main(String[] args) {
        // ========== Working Principal ========== //
        // FIFO --> First In First Out

        // ========== Declaration ========== //
        // Queue is an interface. Usually implemented by LinkedList or ArrayDeque.
        Queue<Integer> que = new LinkedList<>();

        // ========== Adding Elements ========== //
        // Adds 1 at the end of the queue.
        que.add(1);
        que.offer(2); // offer() is preferred as it doesn't throw exceptions on capacity limits

        // ========== Deleting Elements ========== //
        // Removes the front element from the queue and returns it.
        que.poll(); // returns null if empty
        // que.remove(); // throws exception if empty

        // Clear the whole queue.
        que.clear();

        // ========== Accessing Elements ========== //
        // Returns the front element of the queue.
        que.offer(10);
        int front_element = que.peek(); // returns null if empty
        // int front_element = que.element(); // throws exception if empty

        // ========== Other Important Methods ========== //
        // Checks if the queue is empty.
        que.isEmpty(); // Returns true or false.
        // Returns the number of elements in the queue.
        que.size();

        // Iterating Through Queue
        while (!que.isEmpty()) {
            System.out.println(que.peek());
            que.poll();
        }
    }
}

PriorityQueue

import java.util.*;

public class Main {
    public static void main(String[] args) {
        // ========== Working Principle ========== //
        // Min-Heap by default in Java: Smallest element at the top

        // ========== Declaration ========== //
        // Min-Heap (default)
        PriorityQueue<Integer> min_pq = new PriorityQueue<>();

        // Max-Heap (requires Collections.reverseOrder())
        PriorityQueue<Integer> max_pq = new PriorityQueue<>(Collections.reverseOrder());

        // ========== Adding Elements ========== //
        min_pq.add(10);
        min_pq.offer(30);
        min_pq.add(20);

        // ========== Deleting Elements ========== //
        // Removes the top (smallest) element.
        min_pq.poll();
        
        // Clear the entire priority_queue.
        min_pq.clear();

        // ========== Accessing Elements ========== //
        // Returns the top element.
        min_pq.add(10);
        int top_element = min_pq.peek();

        // ========== Iterating Through Priority Queue ========== //
        // NOTE: Direct iteration does not guarantee order!
        // To iterate in sorted order, you must poll elements.
        while (!min_pq.isEmpty()) {
            System.out.print(min_pq.poll() + " ");
        }
        System.out.println();
    }
}

TreeSet (Equivalent to std::set)

import java.util.*;

public class Main {
    public static void main(String[] args) {
        // ========== Note ========== //
        // TreeSet is always sorted in ascending order by default.
        // No duplicates allowed. O(log n) time complexity.

        // ========== Declaration ========== //
        TreeSet<Integer> s = new TreeSet<>();

        // Initialization from another collection
        TreeSet<Integer> s2 = new TreeSet<>(Arrays.asList(5, 20, 60));

        // Descending order TreeSet
        TreeSet<Integer> sDesc = new TreeSet<>(Collections.reverseOrder());

        // ========== Adding New Elements ========== //
        s.add(13);
        s.add(14);

        // ========== Deleting Elements ========== //
        // Delete element from the set.
        s.remove(13);
        
        // Removes all elements.
        s.clear();

        // ========== Accessing Elements ========== //
        s.add(20);
        s.add(30);

        // contains(value) Returns true if value exists
        if (s.contains(20)) {
            System.out.println("Found 20");
        }

        // First (smallest) element
        if (!s.isEmpty()) System.out.println("First: " + s.first());

        // Last (largest) element
        if (!s.isEmpty()) System.out.println("Last: " + s.last());

        // ceiling(e) >= e (lower_bound equivalent)
        System.out.println(s.ceiling(25)); // Returns 30

        // floor(e) <= e
        System.out.println(s.floor(25)); // Returns 20

        // higher(e) > e (upper_bound equivalent)
        System.out.println(s.higher(20)); // Returns 30

        // ========== Other Important Methods ========== //
        s.size();
        s.isEmpty();

        // Iterating set
        for (int num : s) {
            System.out.println(num);
        }
    }
}

TreeMap (Equivalent to std::map)

import java.util.*;

public class Main {
    public static void main(String[] args) {
        // ========== Note ======================== //
        // TreeMap stores key-value pairs
        // Keys are unique and sorted in ascending order by default
        // Average time complexity: O(log n)

        // ========== Declaration ========== //
        TreeMap<Integer, String> m = new TreeMap<>();

        // ========== Adding New Elements ========== //
        m.put(4, "date");
        m.put(5, "elderberry");
        m.put(4, "dragonfruit"); // overwrites "date"

        // ========== Deleting Elements ========== //
        // Erase by key
        m.remove(4);

        // Removes all key-value pairs
        // m.clear();

        // ========== Accessing Elements ========== //
        // Access using key
        String val = m.get(5); // returns null if not found
        
        // getOrDefault
        String val2 = m.getOrDefault(10, "Not Found");

        // containsKey() and containsValue()
        boolean hasKey = m.containsKey(5);
        boolean hasVal = m.containsValue("elderberry");

        // ========== Access First & Last Element ========== //
        if (!m.isEmpty()) {
            System.out.println("First key: " + m.firstKey());
            System.out.println("Last key: " + m.lastKey());
            
            // Get first Entry (Key-Value pair)
            Map.Entry<Integer, String> first = m.firstEntry();
        }

        // lower/higher/floor/ceiling keys
        Integer ceilKey = m.ceilingKey(3); // First key >= 3

        // ========== Iterating over map ========== //
        // Method 1: Iterating over Entry set
        for (Map.Entry<Integer, String> entry : m.entrySet()) {
            System.out.println(entry.getKey() + " => " + entry.getValue());
        }

        // Method 2: Iterating over Keys
        for (int key : m.keySet()) {
            System.out.println(key + " => " + m.get(key));
        }
    }
}

ArrayDeque (Equivalent to Deque)

import java.util.*;

public class Main {
    public static void main(String[] args) {
        // ========== Working Principal ========== //
        // Double Ended Queue — insertion & deletion at both ends
        // Faster than LinkedList when used as queue or stack
        // Average time complexity: O(1)

        // ========== Declaration ========== //
        ArrayDeque<Integer> dq = new ArrayDeque<>();

        // ========== Adding Elements ========== //
        // Adds elements at the back.
        dq.addLast(10);
        dq.offerLast(20);

        // Adds elements at the front.
        dq.addFirst(5);
        dq.offerFirst(1);

        // ========== Deleting Elements ========== //
        // Removes element from the back.
        dq.pollLast(); // Returns null if empty
        // dq.removeLast(); // Throws exception if empty

        // Removes element from the front.
        dq.pollFirst();

        // ========== Accessing Elements ========== //
        int front_element = dq.peekFirst();
        int back_element = dq.peekLast();

        // ========== Iterating Through Deque ========== //
        for (int x : dq) {
            System.out.print(x + " ");
        }
    }
}

Pair (Custom Implementation)

import java.util.*;

public class Main {
    // Java doesn't have a built-in Pair class (except in specific libraries like JavaFX)
    // So we usually create a static nested class for Pair.
    
    static class Pair<U, V> implements Comparable<Pair<U, V>> {
        public final U first;
        public final V second;

        public Pair(U first, V second) {
            this.first = first;
            this.second = second;
        }

        // Optional: Implement compareTo if you need to sort Pairs
        @Override
        public int compareTo(Pair<U, V> o) {
            // Assuming U is Comparable
            return ((Comparable<U>) this.first).compareTo(o.first);
        }
    }

    public static void main(String[] args) {
        // ========== Declaration ========== //
        Pair<Integer, Integer> p1 = new Pair<>(1, 2);
        Pair<Integer, String> p2 = new Pair<>(10, "hello");

        // ========== Accessing Elements ========== //
        System.out.println(p1.first + " " + p1.second);

        // ========== Pair Array / List ========== //
        ArrayList<Pair<Integer, Integer>> vp = new ArrayList<>();
        vp.add(new Pair<>(10, 20));
        vp.add(new Pair<>(5, 15));

        // Sorting custom pair objects
        Collections.sort(vp);
        
        for (Pair<Integer, Integer> pr : vp) {
            System.out.println(pr.first + " " + pr.second);
        }
    }
}

LinkedList (Equivalent to std::list)

import java.util.*;

public class Main {
    public static void main(String[] args) {
        // ========== Working Principle ========== //
        // Doubly linked list — each node points to previous and next.
        // Allows O(1) insertion/deletion anywhere (with iterator).
        // Slower random access O(n) compared to ArrayList.

        // ========== Declaration ========== //
        LinkedList<Integer> l = new LinkedList<>();

        // ========== Adding Elements ========== //
        // Adds to the end of the list.
        l.addLast(10);
        l.add(20); // Default adds to the end
        
        // Adds to the front of the list.
        l.addFirst(5);

        // ========== Deleting Elements ========== //
        // Removes element from the end.
        l.removeLast();
        // Removes element from the front.
        l.removeFirst();

        // ========== Accessing Elements ========== //
        // Access front and back elements
        int first = l.getFirst();
        int last = l.getLast();

        // Slower access via index
        int val = l.get(0);

        // ========== Iterating through List ========== //
        for (int x : l) {
            System.out.print(x + " ");
        }
    }
}

Multi-Set (Frequency Map)

import java.util.*;

public class Main {
    public static void main(String[] args) {
        // ========== Note ========== //
        // Java doesn't have a built-in MultiSet.
        // We simulate it using a TreeMap<Integer, Integer>
        // where Key = element, Value = frequency (count).

        TreeMap<Integer, Integer> ms = new TreeMap<>();

        // ========== Adding Elements ========== //
        int valToInsert = 10;
        ms.put(valToInsert, ms.getOrDefault(valToInsert, 0) + 1);
        
        ms.put(20, ms.getOrDefault(20, 0) + 1);
        ms.put(20, ms.getOrDefault(20, 0) + 1); // Add duplicate

        // ========== Deleting Elements ========== //
        int valToRemove = 20;
        if (ms.containsKey(valToRemove)) {
            int count = ms.get(valToRemove);
            if (count == 1) {
                ms.remove(valToRemove);
            } else {
                ms.put(valToRemove, count - 1);
            }
        }

        // ========== Accessing Elements ========== //
        // Number of occurrences
        int count = ms.getOrDefault(20, 0);

        // First (smallest) element
        if (!ms.isEmpty()) {
            System.out.println("Smallest: " + ms.firstKey());
        }

        // ========== Iterating Multiset ========== //
        for (Map.Entry<Integer, Integer> entry : ms.entrySet()) {
            int element = entry.getKey();
            int frequency = entry.getValue();
            for (int i = 0; i < frequency; i++) {
                System.out.print(element + " ");
            }
        }
        System.out.println();
    }
}
